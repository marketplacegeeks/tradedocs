```python
@pytest.mark.django_db
class TestPlCiPdfEdgeCases:
    """
    Verify the PL+CI PDF generator does not crash on real-world edge-case data:
    - Very long company names/addresses (>100 chars)
    - Missing optional fields (bank, incoterms)
    - Missing product description details

    Risk: ReportLab table cells with unconstrained long strings can overflow the
    page frame and raise LayoutError or silently produce corrupt PDFs.
    Constraint #20: PDF is always in-memory, never written to disk.
    """

        """
        Create a minimal APPROVED PI → PL + CI pair for PDF generation.
        Returns the packing_list instance.
        """
        from apps.commercial_invoice.tests.factories import CommercialInvoiceFactory
        from apps.master_data.tests.factories import (
            BankFactory, IncotermFactory, OrganisationFactory,
        )

        maker = MakerFactory()
        exporter_name = org_name or "Standard Exporter Ltd"
        exporter = OrganisationFactory(name=exporter_name)
        consignee = OrganisationFactory(name=org_name or "Standard Consignee Ltd")

            status="APPROVED",
            exporter=exporter,
            consignee=consignee,
            created_by=maker,
        )

            "proforma_invoice": pi,
            "exporter": exporter,
            "consignee": consignee,
            "status": "DRAFT",
            "created_by": maker,

        pl = PackingListFactory(**pl_kwargs)

            "packing_list": pl,
```

```python
# ---- Concurrent document number generation ----------------------------------

@pytest.mark.django_db(transaction=True)
class TestConcurrentPiCreation:
    """
    Verify that two concurrent POST /proforma-invoices/ requests produce unique
    document numbers (PI-YYYY-NNNN) — the select_for_update() lock in
    generate_document_number() must prevent duplicates.

    Architecture note (NOTES.md): PI numbers are generated inside
    select_for_update() in apps/proforma_invoice/services.py.
    This test exercises that lock under real concurrency.

    Uses transaction=True so each thread gets an independent DB connection
    and real serialization semantics apply (not savepoint-wrapped).
    """

    def _create_pi(self):
        """
        Create a PI via the HTTP API and return the pi_number.
        Uses a fresh client per call (each thread needs its own client).
        """
        from apps.master_data.tests.factories import CurrencyFactory, OrganisationFactory
        maker = MakerFactory()
        client = auth_client(maker)
        payload = {
            "exporter": OrganisationFactory().pk,
            "consignee": OrganisationFactory().pk,
            "currency": CurrencyFactory().pk,
        }
        resp = client.post(PI_LIST_URL, payload, format="json")
        assert resp.status_code == 201, f"PI creation failed: {resp.status_code} {resp.content}"
        return resp.data["pi_number"]

    def test_concurrent_pi_creation_produces_unique_numbers(self):
        """
        Two simultaneous PI creation requests must get different PI numbers.
        If select_for_update() is missing or broken, both would read the same
        counter value and produce duplicate numbers (unique constraint violation
        or silent collision depending on race timing).
        """
        from concurrent.futures import ThreadPoolExecutor, as_completed

        with ThreadPoolExecutor(max_workers=2) as pool:
            futures = [pool.submit(self._create_pi) for _ in range(2)]
            results = [f.result() for f in as_completed(futures)]

        assert len(results) == 2, "Expected 2 PI numbers"
        assert results[0] != results[1], (
            f"Duplicate PI number detected: both threads got {results[0]!r}. "
            "This means select_for_update() in generate_document_number() is "
            "not serializing concurrent requests."
        )
        # Both numbers must follow the PI-YYYY-NNNN format
        for number in results:
            assert number.startswith("PI-"), f"Unexpected PI number format: {number!r}"
```
